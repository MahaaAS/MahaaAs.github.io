function showEmail(){
    var userName="alz.mahaa";
    var emailServer="gmail.com"
    var emailLink = userName + "@" + emailServer;
    userName= stringReverser(userName);
    emailServer= stringReverser(emailServer);
    var mailLink = userName + "@" + emailServer;
    document.write("<a href = mailto:"+emailLink+">"+mailLink+"</a>");  
}

function stringReverser(textString){
 if (!textString)
 return '';
 var revString='';
 for (i= textString.length-1; i>=0; i--)
 revString+=textString.charAt(i);
 return revString;
}
